module.exports = { 
  getApplication: function(req, res) 
  { 
    if (!req.session.user) return res.redirect('/login'); 
 
    Application.find({}, function(err, found){ 
      if(err){ 
        console.log(err); 
      } 
      console.log(found); 
 
      User.findOne({ id: req.session.user}, function (err, user){ 
        if (err) //ra mis erreur 
        { 
          console.log(err); 
        } 
        if (!user) // si l'email n'exuiste pas 
        { 
          console.log('Tsis user.'); //afficher email invalide 
          return res.redirect('/login'); 
        } 
        var retVal = []; 
        retVal['applications'] = found; 
        retVal['users'] = user; 
        res.view( 'user/profile', retVal ); 
      }); 
    }); 
  }, 
};
